%pulls lecture files
\newcommand{\lec}[2]{%
	\foreach \c in {#1,...,#2}{%
		\IfFileExists{Lectures/lec_\c.tex} {%
			\input{Lectures/lec_\c.tex}%
		}{}%
	}%
}

%number you want to call, and b is the highest.
